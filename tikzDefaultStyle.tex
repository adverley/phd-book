\def\defaultNodeVerticalDist{2cm}
\def\defaultSpacedArrow{5pt}

	block/.style={
		rectangle,draw=black,minimum width=8em,text centered,rounded corners,minimum height=4em, node distance=\defaultNodeHorizontalDist and \defaultNodeVerticalDist

			block,fill=ColorMain,thick,inner sep=0pt,minimum size=4mm
	},

	},

    postSpaced/.style={
            post,shorten <=\defaultSpacedArrow,shorten >=\defaultSpacedArrow
    },
    % CONNECTOR STYLE WITH ARROWS

	},
	postArrow/.style={

	},
